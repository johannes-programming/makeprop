Overview

Make a simple property.

Usage


from makeprop import makeprop

# Define a class and use makeprop to create a property
class Example:
    @makeprop()
    def value(self, x):
        """Square the value before storing it."""
        return x ** 2

# Use the property
example = Example()
example.value = 4
print(example.value)  # Output: 16
